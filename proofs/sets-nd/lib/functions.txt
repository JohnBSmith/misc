function_intro. ⊢ relation f →
  (∀x. ∀y1. ∀y2. (x, y1) ∈ f → (x, y2) ∈ f → y1 = y2) →
  function f, axiom.
fn_is_rel. ⊢ function f → relation f, axiom.
fn_unique_value. ⊢ function f → (x, y1) ∈ f → (x, y2) ∈ f → y1 = y2, axiom.
fn_img_lemma. ⊢ function f → x ∈ dom f →
  ∃y. set y ∧ {y} = {y | (x, y) ∈ f}, axiom.
app_intro. ⊢ function f → (x, y) ∈ f → y = app f x, axiom.
app_elim. ⊢ function f → x ∈ dom f → y = app f x → (x, y) ∈ f, axiom.
fn_from_term. ⊢ f = {t | ∃x. t = (x, y x)} → function f, axiom.
